59 // When the display powers up, it is configured as follows:
60 // 0. LCD starts in 8 bit mode
61 // 1. Display clear
62 // 2. Function set:
63 // DL = 1; 8-bit interface data
64 // N = 0; 1-line display
65 // F = 0; 5x8 dot character font
66 // 3. Display on/off control:
67 // D = 0; Display off
68 // C = 0; Cursor off
69 // B = 0; Blinking off
70 // 4. Entry mode set:
71 // I/D = 1; Increment by 1
72 // S = 0; No shift
73 //
74 // Note, however, that resetting the Arduino doesn't reset the LCD, so we
75 // can't assume that its in that state when a application starts (and the
76 // LiquidCrystal constructor is called).
77 // A call to begin() will reinitialize the LCD.
78 //
79 void LCD::begin(uint8_t cols, uint8_t lines, uint8_t dotsize)
80 {
81  if (lines > 1)
82  {
83  _displayfunction |= LCD_2LINE;
84  }
85  _numlines = lines;
86  _cols = cols;
87 
88  // for some 1 line displays you can select a 10 pixel high font
89  // ------------------------------------------------------------
90  if ((dotsize != LCD_5x8DOTS) && (lines == 1))
91  {
92  _displayfunction |= LCD_5x10DOTS;
93  }
94 
95  // SEE PAGE 45/46 FOR INITIALIZATION SPECIFICATION!
96  // according to datasheet, we need at least 40ms after power rises above 2.7V
97  // before sending commands. Arduino can turn on way before 4.5V so we'll wait
98  // 50
99  // ---------------------------------------------------------------------------
100  delay (100); // 100ms delay
101 
102  //put the LCD into 4 bit or 8 bit mode
103  // -------------------------------------
104  if (! (_displayfunction & LCD_8BITMODE))
105  {
106  // this is according to the hitachi HD44780 datasheet
107  // figure 24, pg 46
108 
109  // we start in 8bit mode, try to set 4 bit mode
110  // Special case of "Function Set"
111  send(0x03, FOUR_BITS);
112  delayMicroseconds(4500); // wait min 4.1ms
113 
114  // second try
115  send ( 0x03, FOUR_BITS );
116  delayMicroseconds(150); // wait min 100us
117 
118  // third go!
119  send( 0x03, FOUR_BITS );
120  delayMicroseconds(150); // wait min of 100us
121 
122  // finally, set to 4-bit interface
123  send ( 0x02, FOUR_BITS );
124  delayMicroseconds(150); // wait min of 100us
125 
126  }
127  else
128  {
129  // this is according to the hitachi HD44780 datasheet
130  // page 45 figure 23
131 
132  // Send function set command sequence
133  command(LCD_FUNCTIONSET | _displayfunction);
134  delayMicroseconds(4500); // wait more than 4.1ms
135 
136  // second try
137  command(LCD_FUNCTIONSET | _displayfunction);
138  delayMicroseconds(150);
139 
140  // third go
141  command(LCD_FUNCTIONSET | _displayfunction);
142  delayMicroseconds(150);
143 
144  }
145 
146  // finally, set # lines, font size, etc.
147  command(LCD_FUNCTIONSET | _displayfunction);
148  delayMicroseconds ( 60 ); // wait more
149 
150  // turn the display on with no cursor or blinking default
151  _displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
152  display();
153 
154  // clear the LCD
155  clear();
156 
157  // Initialize to default text direction (for romance languages)
158  _displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
159  // set the entry mode
160  command(LCD_ENTRYMODESET | _displaymode);
161 
162  backlight();
163 
164 }

180 void LCD::setCursor(uint8_t col, uint8_t row)
181 {
182  const byte row_offsetsDef[] = { 0x00, 0x40, 0x14, 0x54 }; // For regular LCDs
183  const byte row_offsetsLarge[] = { 0x00, 0x40, 0x10, 0x50 }; // For 16x4 LCDs
184 
185  if ( row >= _numlines )
186  {
187  row = _numlines-1; // rows start at 0
188  }
189 
190  // 16x4 LCDs have special memory map layout
191  // ----------------------------------------
192  if ( _cols == 16 && _numlines == 4 )
193  {
194  command(LCD_SETDDRAMADDR | (col + row_offsetsLarge[row]));
195  }
196  else
197  {
198  command(LCD_SETDDRAMADDR | (col + row_offsetsDef[row]));
199  }
200 
201 }

293 // Write to CGRAM of new characters
294 void LCD::createChar(uint8_t location, uint8_t charmap[])
295 {
296  location &= 0x7; // we only have 8 locations 0-7
297 
298  command(LCD_SETCGRAMADDR | (location << 3));
299  delayMicroseconds(30);
300 
301  for (uint8_t i = 0; i < 8; i++)
302  {
303  write(charmap[i]); // call the virtual write method
304  delayMicroseconds(40);
305  }
306 }
